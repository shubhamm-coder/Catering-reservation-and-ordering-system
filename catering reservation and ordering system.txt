  <script>
    const inputs = document.querySelectorAll('#orderForm input[type="number"]');
    const totalDisplay = document.querySelector('.total');

    function updateTotal() {
      let total = 0;
      inputs.forEach(input => {
        total += input.value * input.dataset.price;
      });
      totalDisplay.innerText = `Total: ₹${total}`;
    }

    inputs.forEach(input => input.addEventListener('input', updateTotal));

    document.getElementById('reservationForm').addEventListener('submit', function(e) {
      e.preventDefault();
      const name = document.getElementById('name').value;
      const total = totalDisplay.innerText;
      const payment = document.getElementById('paymentMethod').value;
      alert(`Thank you, ${name}! Your booking is confirmed.\n${total}\nPayment Method: ${payment}`);
      this.reset();
      updateTotal();
    });
  </script>
